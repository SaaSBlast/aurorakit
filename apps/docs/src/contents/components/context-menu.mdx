## Usage

```tsx
import {
  ContextMenu,
  ContextMenuContent,
  ContextMenuItem,
  ContextMenuTrigger,
} from '@/components/ui/context-menu'
```

```tsx
<ContextMenu>
  <ContextMenuTrigger>Right click</ContextMenuTrigger>
  <ContextMenuContent>
    <ContextMenuItem>Profile</ContextMenuItem>
    <ContextMenuItem>Billing</ContextMenuItem>
    <ContextMenuItem>Team</ContextMenuItem>
    <ContextMenuItem>Subscription</ContextMenuItem>
  </ContextMenuContent>
</ContextMenu>
```

<Callout emoji="⚠️">
  Unlike `shadcn/ui`, the `inset` prop for `<ContextMenuSubTrigger />
  `, `<ContextMenuItem />` and `<ContextMenuLabel />` is renamed to `insetLeft`, since it collides with
  the css `inset` property.
</Callout>
